\documentclass{article}

\usepackage[toc]{glossaries}
\makeglossaries
%Term definitions	
\newglossaryentry{entryOne}{name=Accuracy, description={The degree of how close a calculated or measured value is to the actual value. (e.g. 3.1416 is more accurate than 3.14 when calculating pi)}}   
\newglossaryentry{entryTwo}{name=Arccos(x), description={The inverse of cosine(x) used to measure the unknown angle of a right tirangle.}}
\newglossaryentry{entryThree}{name=Basic Arithmetics, description={The fundamental mathematical operations used to manipulate and compute numerical values - addition, subtraction, multiplication, and division.}}
\newglossaryentry{entryFour}{name=Command Line, description={An interface for typing commands directly to a computer’s operating system, or as the input to a software.}}
\newglossaryentry{entryFive}{name=Calculation History, description={A record of the previous calculations performed by the user, which can be displayed and recalled for reference or correction purposes.}}
\newglossaryentry{entrySix}{name=Cosine, description={A trigonometric function equal to the ratio of the length of the adjacent side to the hypotenuse of a right triangle.}}
\newglossaryentry{entrySeven}{name=Function, description={A relationship or expression involving one or more variables.}}
\newglossaryentry{entryEight}{name=Gamma Function, description={The gamma function is a mathematical function that extends the factorial function to complex and real numbers.}}
\newglossaryentry{entryNine}{name=Input Field, description={The space to enter numerical values and perform calculations using buttons or a keyboard.}}
\newglossaryentry{entryTen}{name=Logarithmic Function, description={A logarithm is the inverse function of exponentiation that measures the power to which a given base must be raised to produce a given value.}}
\newglossaryentry{entryEleven}{name=Maclauren Series Expansion, description={A representation of a function as an infinite sum of polynomial terms centered at zero, used to approximate the value of the function for a given input within a certain range of accuracy.}}
\newglossaryentry{entryTwelve}{name=Mathematical Expression, description={A combination of numbers, symbols, and operators that can be evaluated or simplified using mathematical rules and operations.}}
\newglossaryentry{entryThirteen}{name=Mathematical Symbol, description={A figure or a combination of figures that is used to represent a mathematical object, an action on mathematical objects, a relation between mathematical objects, or for structuring the other symbols that occur in a formula.}}
\newglossaryentry{entryFourteen}{name=Operation, description={A process in which a number, quantity, expression, etc., is altered or manipulated according to set formal rules, such as those of addition, multiplication and differentiation.}}
\newglossaryentry{entryFifteen}{name=Persona, description={Persona is a profile of our product’s typical customer obtained from the interviews conducted.}}
\newglossaryentry{entrySixteen}{name=Power Function, description={A one-term function with a variable as its base that is raised to a fixed real number.}}
\newglossaryentry{entrySeventeen}{name=Precision, description={The degree of exactness of a calculated or a measured value. (e.g. 4.53678 is as precise as 5.95857)}}
\newglossaryentry{entryEighteen}{name=Project plan, description={A project plan defines project goals and objectives, specifies tasks and how goals will be achieved, identifies what resources will be needed and associated budgets and timelines for completion.}}
\newglossaryentry{entryNineteen}{name=Scientific Calculator, description={A calculator that can compute more than basic algebraic functions. The functions include transcendental functions.}}
\newglossaryentry{entryTwenty}{name=Scientific Notation, description={A floating-point representation system where a number is represented as the product of a number between one and ten and the appropriate power of ten.}}
\newglossaryentry{entryTwentyone}{name=Subordinate function, description={A function that supports one or more of the other functions. (e.g. factorial function or square root function)}}
\newglossaryentry{entryTwentytwo}{name=Syntax, description={A set of rules that govern the order and arrangement of mathematical symbols, in order to create a correct mathematical expression.}}
\newglossaryentry{entryTwentythree}{name=Tooltip, description={A GUI element that provides additional information about a particular item or feature when the user hovers their cursor over it.}}
\newglossaryentry{entryTwentyfour}{name=Transcendental function, description={A function that does not satisfy any single-variable polynomial equation whose coefficients are themselves roots of polynomials.}}
\newglossaryentry{entryTwentyfive}{name=Unit Circle, description={A circle with a radius of one where we can directly measure sine, cosine, and tangent.}}
\newglossaryentry{entryTwentysix}{name=Standard Deviation, description={A measure of dispersion in a frequency distribution, equal to the square root of the mean of the squares of the deviations from the arithmetic mean of the distribution.}}
