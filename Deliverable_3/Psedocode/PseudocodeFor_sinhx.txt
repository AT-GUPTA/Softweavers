Function sinh(x)
	pow1 ← pow(Math.E, x)
	pow2 ← 1/pow1
	result ← (pow1 - pow2)/2
	bigDecimalValue ← create BigDecimal object using Double.toString(result)
	Return bigDecimalValue
End function
